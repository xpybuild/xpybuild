 23          """Return the input flattened to an array. 
 24           
 25          input: any variable composed of lists, strings, lambda functions or other  
 26          objects, nested arbitrarily. 
 27           
 28          Empty strings and None items are removed.  
 29           
 30          Returns a list of strings or other objects depth 1.  
 31   
 32          >>> flatten('hi') 
 33          ['hi'] 
 34          >>> flatten(['hi', 'ho', 'hum']) 
 35          ['hi', 'ho', 'hum'] 
 36          >>> flatten(['hi', ['ho', ['hum'] ] ]) 
 37          ['hi', 'ho', 'hum'] 
 38          >>> flatten(['hi', ('ho', ('hum') ) ]) 
 39          ['hi', 'ho', 'hum'] 
 40          >>> flatten(3) 
 41          [3] 
 42          >>> flatten(lambda: '3') 
 43          ['3'] 
 44          >>> flatten(['hi', lambda: 'ho', 'hum']) 
 45          ['hi', 'ho', 'hum'] 
 46          >>> flatten(None) 
 47          [] 
 48          """ 
 49          if not input: return [] 
 50          if isinstance(input, (list, tuple, set)): 
 51                  rv = [] 
 52                  for l in input: 
 53                          rv = rv + flatten(l) 
 54                  return rv 
 55          elif hasattr(input, 'resolveToString'): # Composeable, etc - delayed 
 56                  return [input] 
 57          elif hasattr(input, '__call__'): # a zero-arg lambda 
 58                  return flatten(input()) 
 59          else: 
 60                  return [input]
